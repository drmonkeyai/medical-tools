function normalizeGender(value: string | null | undefined): GenderValue {
  if (value === "male") return "male";
  if (value === "female") return "female";
  if (value === "other") return "other";
  return "unknown";
}

function getBirthYear(dateOfBirth?: string | null) {
  if (!dateOfBirth) return null;
  const d = new Date(dateOfBirth);
  if (Number.isNaN(d.getTime())) return null;
  return d.getFullYear();
}

function formatCreatedDate(value?: string | null) {
  if (!value) return "â€”";
  const d = new Date(value);
  if (Number.isNaN(d.getTime())) return "â€”";
  return d.toLocaleDateString("vi-VN");
}

function generatePatientCode() {
  const now = new Date();
  const yyyy = now.getFullYear();
  const mm = String(now.getMonth() + 1).padStart(2, "0");
  const dd = String(now.getDate()).padStart(2, "0");
  const random = Math.random().toString(36).slice(2, 8).toUpperCase();

  return `PT-${yyyy}${mm}${dd}-${random}`;
}

function generateCaseCode() {
  const now = new Date();
  const yyyy = now.getFullYear();
  const mm = String(now.getMonth() + 1).padStart(2, "0");
  const dd = String(now.getDate()).padStart(2, "0");
  const random = Math.random().toString(36).slice(2, 8).toUpperCase();

  return `CA-${yyyy}${mm}${dd}-${random}`;
}

function compareText(a: string, b: string) {
  return a.localeCompare(b, "vi");
}

function sortItems(items: CaseListItem[], sortKey: SortKey, sortDirection: SortDirection) {
  const next = [...items];

  next.sort((a, b) => {
    let compare = 0;

    if (sortKey === "case_code") compare = compareText(a.caseCode, b.caseCode);
    if (sortKey === "case_name") compare = compareText(a.caseName, b.caseName);
    if (sortKey === "gender") compare = compareText(formatGender(a.gender), formatGender(b.gender));
    if (sortKey === "birth_year") compare = (a.birthYear ?? 0) - (b.birthYear ?? 0);
    if (sortKey === "created_at") {
      compare =
        (a.createdAt ? new Date(a.createdAt).getTime() : 0) -
        (b.createdAt ? new Date(b.createdAt).getTime() : 0);
    }

    return sortDirection === "asc" ? compare : -compare;
  });

  return next;
}

function calcAge(dateOfBirth?: string | null) {
  if (!dateOfBirth) return null;
  const dob = new Date(dateOfBirth);
  if (Number.isNaN(dob.getTime())) return null;

  const now = new Date();
  let age = now.getFullYear() - dob.getFullYear();
  const monthDiff = now.getMonth() - dob.getMonth();

  if (monthDiff < 0 || (monthDiff === 0 && now.getDate() < dob.getDate())) {
    age -= 1;
  }

  return age >= 0 ? age : null;
}

  async function deleteAssessmentBundle(targetAssessmentId: string) {
    const { data: runRows, error: runError } = await supabase
      .from("calculator_runs")
      .select("id")
      .eq("assessment_id", targetAssessmentId);

      throw runError;
    }

    const runIds = ((runRows ?? []) as { id: string }[]).map((item) => item.id);

    if (runIds.length > 0) {
      const { error: deleteRunInputsError } = await supabase
        .from("calculator_run_inputs")
        .delete()
        .in("calculator_run_id", runIds);

      if (deleteRunInputsError) {
        throw deleteRunInputsError;
      }

      const { error: deleteError } = await supabase
        .from(table)
        .delete()
        .eq("assessment_id", targetAssessmentId);

      if (deleteError) {
        throw deleteError;
      }
    };

    await deleteByAssessmentId("calculator_runs");
    await deleteByAssessmentId("assessment_observations");
    await deleteByAssessmentId("assessment_red_flags");
    await deleteByAssessmentId("assessment_plan_items");

    await deleteByAssessmentId("assessment_notes");
    await deleteByAssessmentId("assessment_vitals");

    const { error: deleteAssessmentError } = await supabase
      .from("case_assessments")
      .delete()
      .eq("id", targetAssessmentId);

    if (deleteAssessmentError) {
      throw deleteAssessmentError;
    }
  }
